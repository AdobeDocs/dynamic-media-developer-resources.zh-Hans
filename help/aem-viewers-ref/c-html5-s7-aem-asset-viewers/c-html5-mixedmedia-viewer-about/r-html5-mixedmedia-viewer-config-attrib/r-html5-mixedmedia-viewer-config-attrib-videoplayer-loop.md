`[VideoPlayer.|<containerId>_videoPlayer.]loop=0|1`

 <tbody> 
  <tr> 
   <td colname="col1"> <p> <span class="codeph"> 0 | 1</span> </p> </td>

  </tr> 
 </tbody>
