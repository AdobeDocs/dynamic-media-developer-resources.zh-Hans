`[VideoPlayer.|<containerId>_videoPlayer.]waiticon=0|1`

<table id="table_C616483932C2482CA9794DDD7313FD7C"> 
 <tbody> 
  <tr> 
   <td colname="col1"> <p> <span class="codeph"> 0 | 1</span> </p> </td>

  </tr> 
 </tbody>
